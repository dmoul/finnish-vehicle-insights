  mutate(model_year = as.numeric(model_year)) |>

data_for_table |>
  arrange(brand, model, desc(model_year)) |>

  tab_options(table.font.size = 10) |>
  sub_missing() |>
  fmt_number(columns = failure_rate,
              decimals = 2)
